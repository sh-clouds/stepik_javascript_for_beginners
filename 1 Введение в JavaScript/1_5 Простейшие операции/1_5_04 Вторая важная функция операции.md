Вторая важная функция операции "+" это приведение к числовому типу. Фактически, действие такое же, как и у функции явного приведения типов - Number(). Если перед значением или переменной поставить знак "+", то значение будет приведено к числу.

let x = false;
console.log(x);     // В консоль будет выведено значение "false"
console.log(+x);    // В консоль будет выведено значение "0"
let x = "100";
let y = 500;
console.log(x+y);   // В консоль будет выведено значение "100500"
console.log(+x+y);  // В консоль будет выведено значение "600"
let x=null;
console.log(x);     // В консоль будет выведено значение "null"
console.log(+x);    // В консоль будет выведено значение "0"

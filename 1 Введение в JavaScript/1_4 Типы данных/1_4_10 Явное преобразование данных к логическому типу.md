Явное преобразование данных к логическому типу производится функцией Boolean().
Тут все еще проще - любые данные, в которых есть какое-то значение, превращаются в true, любые данные, обозначающие отсутствие значения (пустая строка, число ноль, значение null, NaN или Undefined) превращаются в false:

Пример со значением "пустая строка":

let x = "";             // Создаем переменную x и помещаем в нее пустую строку, тип данных в ней - string.
console.log(typeof x)   // Проверяем тип данных в переменной x - результат будет "string"
let y = Boolean(x);     // Создаем переменную y и присваиваем ей значение из функции Boolean() с аргументом x
console.log(typeof y);  // Проверяем тип данных в переменной y, путем вывода в консоль типа данных, там будет 
                        // значение "boolean". При этом значение переменной y будет выглядеть как false.
 

И пример с числом:

let x = 100;            // Создаем переменную со значением число 100, тип данных в ней - number.
console.log(typeof x)   // Проверяем тип данных в переменной x - результат будет "number"
let y = Boolean(x);     // Создаем переменную y и присваиваем ей значение из функции Boolean() с аргументом x
console.log(typeof y);  // Проверяем тип данных в переменной y, путем вывода в консоль типа данных, там будет 
                        // значение "boolean". При этом значение переменной y будет выглядеть как "true".

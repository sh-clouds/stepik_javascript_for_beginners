Явное преобразование данных к строковому типу производится функцией String().
Тут все просто - любые данные, являющиеся аргументом этой функции превращаются в строку.

Пример с числом:

let x = 100;              // Создаем переменную со значением 100, тип данных в ней - числовой.
console.log(typeof x)     // Проверяем тип данных в переменной x - результат будет "number"
let y = String(x);        // Создаем переменную y и присваиваем ей значение из функции String()
                          // с аргументом x
console.log(typeof y);    // Проверяем тип данных в переменной y, путем вывода в консоль типа данных,
                          // там будет значение "string". При этом значение переменной y будет "100".
Теперь приведем пример с приведением логических данных к строковому типу. Тут тоже все просто - логическое значение false превращается в строку "false", логическое значение true превращается в строку "true":

let x = false;            // Создаем переменную со значением false, тип данных в ней - логический, boolean.
console.log(typeof x)     // Проверяем тип данных в переменной x - результат будет "boolean"
let y = String(x);        // Создаем переменную y и присваиваем ей значение из функции String()
                          // с аргументом x
console.log(typeof y);    // Проверяем тип данных в переменной y, путем вывода в консоль типа данных, 
                          // там будет значение "string". При этом значение переменной y будет "false".
Данные, имеющие значения null, NaN или Undefined превращаются в текстовые выражения "null", "NaN" и "Undefined".

let x;                    // Создаем переменную не устанавливая ей значение.
console.log(typeof x)     // Проверяем тип данных в переменной x - результат будет "undefined"
let y = String(x);        // Создаем переменную y и присваиваем ей значение из функции String()
                          // с аргументом x
console.log(typeof y);    // Проверяем тип данных в переменной y, путем вывода в консоль типа данных, 
                          // там будет значение "string". При этом значение переменной y будет "undefined".

В этом задании вам нужно вычислить сумму всех четных чисел, встречающихся в ряду от 1 до числа (включительно), передаваемого в нашу функцию (переменная "а").

Sample Input:
3

Sample Output:
2

function testWhile(a) {
    var x=0;
    let i = 1;
    // Тут нужно написать решение
    while (i<=a)
    {
        if (i%2 == 0)
        {
            x += i;
        }
        i++;
    }
    return x;
}
